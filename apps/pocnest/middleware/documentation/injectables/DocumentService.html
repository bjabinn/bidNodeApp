import { Injectable, HttpService } from '@nestjs/common';
import { Observable } from 'rxjs';
import { DocumentBuilder } from '../builder/document.builder';

@Injectable()
export class DocumentService {
    constructor(private http: HttpService) {}

    getStep(id): Observable<any> {
        return this.http.get(
            `http://virtserver.swaggerhub.com/lperezga/lpl1la/1.0.0/steps/${id}`
        );
    }
    postStep(id, body): Observable<any> {
        return this.http.post(
            `http://virtserver.swaggerhub.com/lperezga/lpl1la/1.0.0/steps/${id}/move-next`,
            body
        );
    }
}
