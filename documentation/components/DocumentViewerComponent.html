import { Component, OnInit, Input } from '@angular/core';
import { DocumentModel } from '@bid/pp/app/core/models/document.model';
import { BehaviorSubject } from 'rxjs';
import { FileRestrictions } from '@progress/kendo-angular-upload';

@Component({
    selector: 'bid-document-viewer',
    templateUrl: './document-viewer.component.html'
})
export class DocumentViewerComponent implements OnInit {
    @Input() documentCollection: BehaviorSubject<DocumentModel[]>;
    @Input() documentRestriction: FileRestrictions;
    @Input() formatDate: string;
    @Input() limit: number;
    @Input() readonly: boolean;

    constructor() {}

    ngOnInit(): void {}
}

<section class="container-fluid">
    <div class="row">
        <div class="col-12">
            <div class="panelbar-wrapper">
                <kendo-panelbar>
                    <kendo-panelbar-item [title]="'DOCUMENTS'" expanded="true">
                        <ng-template kendoPanelBarContent>
                            <div
                                *ngIf="
                                    (documentCollection | async)?.length > 0;
                                    else uploadZone

                                        let doc of documentCollection

                                    [documentRestriction]="documentRestriction"
                                    [formatDate]="formatDate"
                                    [readonly]="readonly"

                                    *ngIf="
                                        limit >
                                        (documentCollection | async).length
                                    "

                                    ></ng-container>
                                </div>
