import { MeetingDto } from '../dtos/meeting.dto'; 
import { Meeting } from '../models/meeting.model'; 
import { MeetingDesignationMapper } from './meeting-designation.mapper'; 
import { DistributionListMapper } from './distribution-list.mapper'; 
import { CirculationPeriodMapper } from './circulation-period.mapper'; 
import { FaceToFaceMeetingMapper } from './face-to-face-meeting.mapper'; 

export class MeetingMapper {
    static ConvertToModel(dto: MeetingDto) : Meeting{
        var model = { 
            id: dto.id,
            name: MeetingDesignationMapper.ConvertToModel(dto.name),
            distributionList: DistributionListMapper.ConvertToModel(dto.distributionList),
            circulationPeriod: CirculationPeriodMapper.ConvertToModel(dto.circulationPeriod),
            faceToFaceMeeting: FaceToFaceMeetingMapper.ConvertToModel(dto.faceToFaceMeeting)
        };
        
        return model;
    }
    
    static ConvertToModels(dtos: MeetingDto[]) : Meeting[]{
        return dtos.map(dto => MeetingMapper.ConvertToModel(dto));
    }
}
